const SmoothEquityCurve = () => {
  const data = [
    { time: '09:00', value: 42850 },
    { time: '09:30', value: 43120 },
    { time: '10:00', value: 42980 },
    { time: '10:30', value: 43340 },
    { time: '11:00', value: 43180 },
    { time: '11:30', value: 43560 },
    { time: '12:00', value: 43490 },
    { time: '12:30', value: 43720 },
    { time: '13:00', value: 43650 },
    { time: '13:30', value: 43910 },
    { time: '14:00', value: 43780 },
    { time: '14:30', value: 44050 },
    { time: '15:00', value: 43970 },
    { time: '15:30', value: 44210 },
    { time: '16:00', value: 44150 },
    { time: '16:30', value: 44380 },
    { time: '17:00', value: 43992 },
  ];

  const [hovered, setHovered] = React.useState(null);
  const [animated, setAnimated] = React.useState(false);
  const svgRef = React.useRef(null);

  React.useEffect(() => {
    const t = setTimeout(() => setAnimated(true), 100);
    return () => clearTimeout(t);
  }, []);

  const W = 1000;
  const H = 280;
  const PAD = { top: 20, right: 10, bottom: 10, left: 10 };

  const minVal = Math.min(...data.map(d => d.value)) - 100;
  const maxVal = Math.max(...data.map(d => d.value)) + 100;

  const xOf = (i) => PAD.left + (i / (data.length - 1)) * (W - PAD.left - PAD.right);
  const yOf = (v) => PAD.top + (1 - (v - minVal) / (maxVal - minVal)) * (H - PAD.top - PAD.bottom);

  const toPath = (pts) => {
    if (pts.length < 2) return '';
    let d = `M ${pts[0].x} ${pts[0].y}`;
    for (let i = 0; i < pts.length - 1; i++) {
      const cp1x = pts[i].x + (pts[i + 1].x - pts[i].x) * 0.5;
      const cp1y = pts[i].y;
      const cp2x = pts[i].x + (pts[i + 1].x - pts[i].x) * 0.5;
      const cp2y = pts[i + 1].y;
      d += ` C ${cp1x} ${cp1y}, ${cp2x} ${cp2y}, ${pts[i + 1].x} ${pts[i + 1].y}`;
    }
    return d;
  };

  const pts = data.map((d, i) => ({ x: xOf(i), y: yOf(d.value) }));
  const linePath = toPath(pts);
  const areaPath = linePath + ` L ${pts[pts.length - 1].x} ${H} L ${pts[0].x} ${H} Z`;

  const handleMouseMove = (e) => {
    if (!svgRef.current) return;
    const rect = svgRef.current.getBoundingClientRect();
    const mouseX = ((e.clientX - rect.left) / rect.width) * W;
    let closest = 0;
    let minDist = Infinity;
    pts.forEach((p, i) => {
      const dist = Math.abs(p.x - mouseX);
      if (dist < minDist) { minDist = dist; closest = i; }
    });
    setHovered(closest);
  };

  const gridValues = [minVal, minVal + (maxVal - minVal) * 0.25, minVal + (maxVal - minVal) * 0.5, minVal + (maxVal - minVal) * 0.75, maxVal];

  const pnl = hovered !== null ? data[hovered].value - data[0].value : data[data.length - 1].value - data[0].value;
  const pnlPct = ((pnl / data[0].value) * 100).toFixed(2);
  const isUp = pnl >= 0;

  return (
    <div className="relative w-full mt-2">
      <div className="flex items-center gap-6 mb-5 px-1">
        <div>
          <p className="text-[9px] text-slate-600 uppercase tracking-widest font-black mb-0.5">Current</p>
          <p className="text-xl font-black text-white font-mono">
            ${(hovered !== null ? data[hovered].value : data[data.length - 1].value).toLocaleString()}
          </p>
        </div>
        <div className={`flex items-center gap-1.5 px-3 py-1.5 rounded-md border ${isUp ? 'bg-emerald-500/10 border-emerald-500/20' : 'bg-red-500/10 border-red-500/20'}`}>
          <svg width="10" height="10" viewBox="0 0 10 10">
            <path d={isUp ? "M5 1 L9 9 L1 9 Z" : "M5 9 L9 1 L1 1 Z"} fill={isUp ? '#10b981' : '#ef4444'} />
          </svg>
          <span className={`text-sm font-black font-mono ${isUp ? 'text-emerald-400' : 'text-red-400'}`}>
            {isUp ? '+' : ''}${pnl.toFixed(0)} ({isUp ? '+' : ''}{pnlPct}%)
          </span>
        </div>
        <div className="ml-auto text-[9px] text-slate-600 font-mono uppercase tracking-widest">
          {hovered !== null ? data[hovered].time : data[data.length - 1].time}
        </div>
      </div>

      <div className="relative">
        <svg
          ref={svgRef}
          viewBox={`0 0 ${W} ${H}`}
          className="w-full"
          style={{ height: '280px' }}
          preserveAspectRatio="none"
          onMouseMove={handleMouseMove}
          onMouseLeave={() => setHovered(null)}
        >
          <defs>
            <linearGradient id="areaGrad" x1="0" y1="0" x2="0" y2="1">
              <stop offset="0%" stopColor="#6366f1" stopOpacity="0.35" />
              <stop offset="55%" stopColor="#6366f1" stopOpacity="0.08" />
              <stop offset="100%" stopColor="#6366f1" stopOpacity="0" />
            </linearGradient>
            <linearGradient id="lineGrad" x1="0" y1="0" x2="1" y2="0">
              <stop offset="0%" stopColor="#818cf8" />
              <stop offset="100%" stopColor="#a78bfa" />
            </linearGradient>
            <filter id="glow">
              <feGaussianBlur stdDeviation="3" result="blur" />
              <feMerge><feMergeNode in="blur"/><feMergeNode in="SourceGraphic"/></feMerge>
            </filter>
            <clipPath id="chartClip">
              <rect x="0" y="0" width={animated ? W : 0} height={H} style={{ transition: 'width 1.4s cubic-bezier(0.4,0,0.2,1)' }} />
            </clipPath>
          </defs>

          {gridValues.map((v, i) => (
            <g key={i}>
              <line x1={PAD.left} y1={yOf(v)} x2={W - PAD.right} y2={yOf(v)} stroke="#1e293b" strokeWidth="1" strokeDasharray={i === 0 || i === 4 ? 'none' : '4 6'} />
              <text x={PAD.left - 4} y={yOf(v) + 4} textAnchor="end" fontSize="9" fill="#334155" fontFamily="monospace">
                {Math.round(v / 100) * 100 >= 1000 ? `${(Math.round(v / 100) * 100 / 1000).toFixed(1)}k` : Math.round(v / 100) * 100}
              </text>
            </g>
          ))}

          <path d={areaPath} fill="url(#areaGrad)" clipPath="url(#chartClip)" />
          <path d={linePath} fill="none" stroke="url(#lineGrad)" strokeWidth="2.5" strokeLinecap="round" strokeLinejoin="round" filter="url(#glow)" clipPath="url(#chartClip)" />

          {hovered !== null && (
            <>
              <line x1={pts[hovered].x} y1={PAD.top} x2={pts[hovered].x} y2={H - PAD.bottom} stroke="#6366f1" strokeWidth="1" strokeDasharray="4 4" opacity="0.5" />
              <line x1={PAD.left} y1={pts[hovered].y} x2={W - PAD.right} y2={pts[hovered].y} stroke="#6366f1" strokeWidth="1" strokeDasharray="4 4" opacity="0.3" />
              <circle cx={pts[hovered].x} cy={pts[hovered].y} r="10" fill="#6366f1" opacity="0.15" />
              <circle cx={pts[hovered].x} cy={pts[hovered].y} r="5" fill="#0f172a" stroke="#818cf8" strokeWidth="2.5" />
              <circle cx={pts[hovered].x} cy={pts[hovered].y} r="2.5" fill="#a78bfa" />
            </>
          )}

          {hovered === null && (
            <>
              <circle cx={pts[pts.length - 1].x} cy={pts[pts.length - 1].y} r="8" fill="#6366f1" opacity="0.2" />
              <circle cx={pts[pts.length - 1].x} cy={pts[pts.length - 1].y} r="4" fill="#0f172a" stroke="#818cf8" strokeWidth="2.5" />
              <circle cx={pts[pts.length - 1].x} cy={pts[pts.length - 1].y} r="2" fill="#a78bfa" />
            </>
          )}
        </svg>
      </div>

      <div className="flex justify-between mt-2 px-1">
        {['09:00', '11:00', '13:00', '15:00', '17:00'].map(t => (
          <span key={t} className="text-[10px] font-medium text-slate-600 font-mono">{t}</span>
        ))}
      </div>
    </div>
  );
};
