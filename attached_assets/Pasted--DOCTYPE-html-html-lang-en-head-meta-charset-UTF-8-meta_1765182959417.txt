    <script>
        const menuToggle = document.getElementById('menu-toggle');
        const menuCloseBtn = document.getElementById('menu-close-btn');
        const mobileMenuPanel = document.getElementById('mobile-menu-panel');
        const menuIconOpen = document.getElementById('menu-icon-open');
        const menuIconClose = document.getElementById('menu-icon-close');

        function toggleMobileMenu() {
            const isOpen = mobileMenuPanel.classList.toggle('open');
            // Toggle the menu button icon
            menuIconOpen.classList.toggle('hidden', isOpen);
            menuIconClose.classList.toggle('hidden', !isOpen);
            // Accessibility: Prevent body scrolling when menu is open
            document.body.classList.toggle('overflow-hidden', isOpen);
        }

        // Event listeners
        menuToggle.addEventListener('click', toggleMobileMenu);
        menuCloseBtn.addEventListener('click', toggleMobileMenu);

        // Close menu on screen resize (if it transitions to desktop view)
        window.addEventListener('resize', () => {
            if (window.innerWidth >= 1024) { // 1024px is the 'lg' breakpoint in Tailwind
                if (mobileMenuPanel.classList.contains('open')) {
                    toggleMobileMenu();
                }
            }
        });

        // --- TRADING SESSION CLOCK LOGIC ---

        // Trading sessions defined in 24-hour UTC time
        const SESSIONS = [
            // Note: Times are rounded to the nearest hour for standard session definitions
            { name: 'Sydney', start: 22, end: 7, nextDayEnd: true }, // 22:00 to 07:00 UTC
            { name: 'Tokyo', start: 0, end: 9 }, // 00:00 to 09:00 UTC
            { name: 'London', start: 7, end: 16 }, // 07:00 to 16:00 UTC
            { name: 'New York', start: 12, end: 21 } // 12:00 to 21:00 UTC
        ];

        function updateSessionClock() {
            const utcTimeElement = document.getElementById('utc-time');
            const sessionStatusElement = document.getElementById('session-status');
            if (!utcTimeElement || !sessionStatusElement) return;

            const now = new Date();
            const utcHours = now.getUTCHours();
            const utcMinutes = now.getUTCMinutes();
            const utcSeconds = now.getUTCSeconds();
            const utcTimeInMinutes = utcHours * 60 + utcMinutes;

            let activeSession = null;
            let timeInMinutes = 0;

            // 1. Update UTC Clock
            const formatTime = (unit) => String(unit).padStart(2, '0');
            utcTimeElement.textContent = `${formatTime(utcHours)}:${formatTime(utcMinutes)}:${formatTime(utcSeconds)}`;

            // 2. Determine Active Session
            for (const session of SESSIONS) {
                let startTimeInMinutes = session.start * 60;
                let endTimeInMinutes = session.end * 60;

                let isActive = false;

                if (session.nextDayEnd) {
                    // Session spans midnight (e.g., Sydney)
                    if (utcTimeInMinutes >= startTimeInMinutes || utcTimeInMinutes < endTimeInMinutes) {
                        isActive = true;
                        if (utcTimeInMinutes >= startTimeInMinutes) {
                            timeInMinutes = utcTimeInMinutes - startTimeInMinutes;
                        } else {
                            // Calculates time from yesterday's start until now
                            const minutesInDay = 24 * 60;
                            timeInMinutes = (minutesInDay - startTimeInMinutes) + utcTimeInMinutes;
                        }
                    }
                } else {
                    // Session within the same day
                    if (utcTimeInMinutes >= startTimeInMinutes && utcTimeInMinutes < endTimeInMinutes) {
                        isActive = true;
                        timeInMinutes = utcTimeInMinutes - startTimeInMinutes;
                    }
                }

                if (isActive) {
                    activeSession = session.name;
                    break;
                }
            }

            // 3. Update Session Status Display
            if (activeSession) {
                const hoursIn = Math.floor(timeInMinutes / 60);
                const minutesIn = timeInMinutes % 60;
                
                let timeString = '';
                if (hoursIn > 0) {
                    timeString += `${hoursIn}h `;
                }
                timeString += `${minutesIn}m`; // Always show minutes

                sessionStatusElement.innerHTML = `Session: <span class="font-bold text-emerald-600">${activeSession}</span> (<span class="font-bold text-emerald-600">${timeString}</span> In)`;
            } else {
                // Quiet Period - calculate time until next major session (Sydney starts at 22:00 UTC)
                const nextStartTimeInMinutes = SESSIONS.find(s => s.name === 'Sydney').start * 60;
                
                // Calculate time until 22:00 UTC
                let timeUntilNext = nextStartTimeInMinutes - utcTimeInMinutes;
                if (timeUntilNext < 0) {
                    // If timeUntilNext is negative, the session is already tomorrow. Wrap around 24 hours.
                    timeUntilNext += (24 * 60);
                }

                const hoursUntil = Math.floor(timeUntilNext / 60);
                const minutesUntil = timeUntilNext % 60;
                
                let timeString = '';
                if (hoursUntil > 0) {
                    timeString += `${hoursUntil}h `;
                }
                timeString += `${minutesUntil}m`;

                sessionStatusElement.innerHTML = `Session: <span class="font-bold text-gray-500">Quiet</span> (Next in ${timeString})`;
            }
        }

        // Run immediately and set interval to update every second
        updateSessionClock();
        setInterval(updateSessionClock, 1000); 
    </script>
